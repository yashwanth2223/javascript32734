const person={
"id":101,
"name":"klu",
"age":30,
"place":"guntur",
}
document.writeln(typeof person);
document.writeln(person.id);
document.writeln(person.name);
document.writeln(person.age);
document.writeln(person.place);
document.writeln(Object.keys(person));
document.writeln(Object.values(person));
document.writeln(Object.entries(person));
document.writeln("person object keys");
for(let key of Object.keys(person))
{
document.writeln(key);
}
document.writeln("person object values");
for(let value of Object.values(person))
{
document.writeln(value);
}
document.writeln("person object keys&values");
for(const [k,v] of Object.entries(person))
{
document.writeln(k,v);
}
for(const key of Object.keys(person))
{
document.writeln(person[key]);
}
